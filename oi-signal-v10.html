Единый прокси для T-Invest + MOEX AlgoPack
Один Cloudflare Worker проксирует оба API. Настройка — 3 минуты.

1. workers.cloudflare.com → Start for free → Create Worker → Edit code
2. Замени весь код на: 
export default {
  async fetch(request) {
    const url = new URL(request.url);
    const path = url.pathname + url.search;
    if (request.method === 'OPTIONS')
      return new Response(null, { headers: cors() });

    const auth = request.headers.get('Authorization') || '';
    let target, hdrs;

    if (path.startsWith('/iss/')) {
      // MOEX AlgoPack
      target = 'https://apim.moex.com' + path;
      hdrs = { 'Authorization': auth, 'Accept': 'application/json' };
    } else {
      // T-Invest REST API
      target = 'https://invest-public-api.tinkoff.ru/rest' + path;
      hdrs = { 'Authorization': auth, 'Content-Type': request.headers.get('Content-Type') || 'application/json' };
    }

    const body = request.method !== 'GET' ? await request.arrayBuffer() : undefined;
    const resp = await fetch(target, { method: request.method, headers: hdrs, body });
    return new Response(await resp.arrayBuffer(), {
      status: resp.status,
      headers: { ...cors(), 'Content-Type': resp.headers.get('Content-Type') || 'application/json' }
    });
  }
};
function cors() {
  return {
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
    'Access-Control-Allow-Headers': 'Authorization, Content-Type',
  };
}
3. Save & Deploy → скопируй URL (вида https://xxx.workers.dev)
4. Вставь URL в поле "прокси" выше → Enter
